355  inline void EvaluateBiomassComponent(int year, int season) {
356 
357 
358  REAL_T o = this->survey_biomass_data->get_error(year, season);
359  REAL_T temp = this->survey_biomass_data->get(year, season);
360  if (temp != this->survey_biomass_data->missing_value) {
361  this->survey_biomass_component +=
362  (.5) * atl::pow((std::log(temp + o) -
363  atl::log(this->survey_biomass_total[year * seasons + season] + o)
364  + (std::pow(this->srv_CV, 2.0) / 2.0)) / this->srv_CV, 2.0);

368  inline void EvaluateAgeCompComponent(int year, int season) {
369  REAL_T o = .00001;
370  variable sum = 0.0;
371 #warning hardcoded survey age comp sample size
372  // REAL_T temp_sample_size = this->survey_proportion_at_age_data_N->get(year, season);
373  REAL_T temp_sample_size = static_cast<REAL_T> (10.0);
374  if (temp_sample_size != this->survey_proportion_at_age_data->missing_value) {
375  for (int a = 0; a <this->ages; a++) {
376  size_t index = year * this->seasons * this->ages + (season) * this->ages + a;
377  REAL_T temp_data = this->survey_proportion_at_age_data->get(year, season, a);
378  o = this->survey_proportion_at_age_data->get_error(year, season, a);
379  sum -= temp_sample_size * temp_data *
380  (atl::log(this->survey_proportion_at_age[index] + o) - std::log(temp_data + o));
381  }
382  }
383 
384  this->survey_age_comp_component -= sum;
385  }
